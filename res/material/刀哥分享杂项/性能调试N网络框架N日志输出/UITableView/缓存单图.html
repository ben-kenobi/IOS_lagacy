/// 缓存指定数组中的单张图片
private func cacheSingleImage(array: [HMStatusViewModel]) {

    // 0. 定义调度组
    let group = dispatch_group_create()
    var dataLength = 0

    // 1. 遍历数组获取单张图片
    for vm in array {

        let urls = vm.status?.retweeted_status?.pic_urls ?? vm.status?.pic_urls

        if urls?.count != 1 {
            continue
        }

        // 2. 利用 SDWebImage 缓存单张图片
        let url = urls![0].picUrl!

        // 1> 入组
        dispatch_group_enter(group)
        SDWebImageManager.sharedManager().downloadImageWithURL(url, options: [], progress: nil) { (image, _, _, _, _) -> Void in

            // 2> 记录缓存图片的长度(已经缓存过的同样会包含)
            if let webImage = image, data = UIImagePNGRepresentation(webImage) {
                dataLength += data.length
            }

            // 3> 出组
            dispatch_group_leave(group)
        }
    }

    // 3. 监听调度组
    dispatch_group_notify(group, dispatch_get_main_queue()) {
        print("缓存图片长度 \(dataLength / 1024) K")
    }
}

/// 缓存指定数组中的单张图片
private func cacheSingleImage(array: [HMStatusViewModel], complete: (isSuccess: Bool, count: Int)->()) {

    // 0. 定义调度组
    let group = dispatch_group_create()
    var dataLength = 0

    // 1. 遍历数组获取单张图片
    for vm in array {

        let urls = vm.status?.retweeted_status?.pic_urls ?? vm.status?.pic_urls

        if urls?.count != 1 {
            continue
        }

        // 2. 利用 SDWebImage 缓存单张图片
        let url = urls![0].picUrl!

        // 1> 入组
        dispatch_group_enter(group)
        SDWebImageManager.sharedManager().downloadImageWithURL(url, options: [], progress: nil) { (image, _, _, _, _) -> Void in

            // 2> 记录缓存图片的长度(已经缓存过的同样会包含)
            if let webImage = image, data = UIImagePNGRepresentation(webImage) {
                dataLength += data.length
            }

            // 3> 出组
            dispatch_group_leave(group)
        }
    }

    // 3. 监听调度组
    dispatch_group_notify(group, dispatch_get_main_queue()) {
        print("缓存图片长度 \(dataLength / 1024) K")

        complete(isSuccess: true,count: array.count)
    }
}

/// 设置图片 url 和 尺寸
///
/// - parameter urls: 配图 URL 数组
/// - parameter size: 配图视图尺寸
func setPicurls(urls: [HMStatusPictureInfo]?, viewSize: CGSize) {

    pic_urls = urls

    countLabel.text = "\(pic_urls?.count ?? 0)"

    let size = urls == nil ? CGSizeZero : viewSize

    /// 单张图片处理 layout 大小
    let layout = collectionViewLayout as! UICollectionViewFlowLayout
    layout.itemSize = urls?.count == 1 ? viewSize : CGSize(width: itemWH, height: itemWH)

    self.snp_updateConstraints { (make) -> Void in
        self.sizeConstraint = make.size.equalTo(size).constraint
    }

    reloadData()
}

    let minWidth:CGFloat = 40
    let maxWidth:CGFloat = 300

    var size = image.size

    // 1. 过窄处理
    size.width = size.width < minWidth ? minWidth : size.width
    // 2. 过宽处理
    if size.width > maxWidth {
        size.width = maxWidth
        size.height = image.size.height * size.width / image.size.width

    // 设置图像视图大小
    pictureViewSize = size
}
